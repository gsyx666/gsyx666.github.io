    results = gg.getResults(resultCounts)

    --遍历结果集

    for i=1 , resultCounts do

    --初始化DEX起始地址

    startAddr = results[i].address - 0x28

    --校验指定偏移特征是否正确，正确第话基本上就确定上dex了

            --正确的dex文件 读取fileSize(偏移量)
